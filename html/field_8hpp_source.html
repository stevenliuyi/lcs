355  void InterpolateFrom(Velocity<T, Dim>& ref_vel)
356  {
357  // interpolation only works for orthogonal coordinates
358  auto const ref_pos_x = ref_vel.GetPosition().GetRange(0);
359  auto const ref_pos_y = ref_vel.GetPosition().GetRange(1);
360 
361  #pragma omp parallel for
362  for (unsigned i = 0; i < this->nx_; ++i)
363  {
364  T pos_x, pos_y;
365  Tensor<Vector<T, 2>, 2> ref_v(2,2);
366 
367  for (unsigned j = 0; j < this->ny_; ++j)
368  {
369  // skip if the current position is out of bound
370  if (pos_.IsOutOfBound(i,j)) continue;
371 
372  std::tie(pos_x, pos_y) = pos_.Get(i, j);
373 
374  auto iter_x = std::upper_bound(ref_pos_x.begin(), ref_pos_x.end(), pos_x);
375  if (iter_x == ref_pos_x.end()) --iter_x;
376  if (iter_x == ref_pos_x.begin()) ++iter_x;
377  int i_next = std::distance(ref_pos_x.begin(), iter_x);
378  int i_pre = i_next - 1;
379 
380  auto iter_y = std::upper_bound(ref_pos_y.begin(), ref_pos_y.end(), pos_y);
381  if (iter_y == ref_pos_y.end()) --iter_y;
382  if (iter_y == ref_pos_y.begin()) ++iter_y;
383  int j_next = std::distance(ref_pos_y.begin(), iter_y);
384  int j_pre = j_next - 1;
385 
386  // add another getter for Tensor class?
387  std::tie(ref_v(0,0).x, ref_v(0,0).y) = ref_vel.Get(i_pre, j_pre);
388  std::tie(ref_v(0,1).x, ref_v(0,1).y) = ref_vel.Get(i_pre, j_next);
389  std::tie(ref_v(1,0).x, ref_v(1,0).y) = ref_vel.Get(i_next, j_pre);
390  std::tie(ref_v(1,1).x, ref_v(1,1).y) = ref_vel.Get(i_next, j_next);
391 
392  auto vx1 = interpolate(ref_pos_x[i_pre], ref_pos_x[i_next],
393  ref_v(0,0).x, ref_v(1,0).x, pos_x);
394  auto vx2 = interpolate(ref_pos_x[i_pre], ref_pos_x[i_next],
395  ref_v(0,1).x, ref_v(1,1).x, pos_x);
396  auto vxm = interpolate(ref_pos_y[j_pre], ref_pos_y[j_next],
397  vx1, vx2, pos_y);
398 
399  auto vy1 = interpolate(ref_pos_x[i_pre], ref_pos_x[i_next],
400  ref_v(0,0).y, ref_v(1,0).y, pos_x);
401  auto vy2 = interpolate(ref_pos_x[i_pre], ref_pos_x[i_next],
402  ref_v(0,1).y, ref_v(1,1).y, pos_x);
