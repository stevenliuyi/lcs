280  inline void SetCurrentVelocity()
281  {
282  T signed_data_delta = (this->direction_ == Forward) ? data_delta_ : (-data_delta_);
283 
284  // initialize data velocities
285  if (this->current_time_ == this->initial_time_)
286  {
287  current_data_time_ = begin_data_time_;
288 
289  // find the data velocity field that is temporally closest to the initial time
290  switch(this->direction_)
291  {
292  case Forward:
293  while (current_data_time_ < this->initial_time_)
294  current_data_time_ += signed_data_delta;
295  break;
296  case Backward:
297  while (current_data_time_ > this->initial_time_)
298  current_data_time_ -= signed_data_delta;
299  break;
300  default: break;
301  }
302 
303  // read two temporally adjacent data field (for later interpolation)
304  #pragma omp parallel
305  #pragma omp single nowait
306  {
307  #pragma omp task
308  previous_data_vel_->UpdateTime(current_data_time_);
309  ReadDataVelocityFromFile(*previous_data_vel_);
310  #pragma omp task
311  next_data_vel_->UpdateTime(current_data_time_ + signed_data_delta);
312  ReadDataVelocityFromFile(*next_data_vel_);
313  }
314 
315  // check if the current time is outside the time
316  // range of two temporally adjacent data field
317  } else if (((this->current_time_ >= current_data_time_ + signed_data_delta) &&
318  (end_data_time_ > current_data_time_ + signed_data_delta) &&
319  this->direction_ == Forward) ||
320  ((this->current_time_ <= current_data_time_ + signed_data_delta) &&
321  (end_data_time_ < current_data_time_ + signed_data_delta) &&
322  this->direction_ == Backward))
323  {
324  // update data velocities
325 
326  current_data_time_ += signed_data_delta;
327 
328  // read the next two temporally adjacent data velocity field so that
329  // the current time could be within the time range of the two data field
330  #pragma omp parallel
331  #pragma omp single nowait
332  {
333  #pragma omp task
334  previous_data_vel_->UpdateTime(current_data_time_);
335  ReadDataVelocityFromFile(*previous_data_vel_);
336 
337  #pragma omp task
338  next_data_vel_->UpdateTime(current_data_time_ + signed_data_delta);
339  ReadDataVelocityFromFile(*next_data_vel_);
340  }
341  }
342 
343  // temporal interpolation
344  interpolate(previous_data_vel_->GetTime(),
345  next_data_vel_->GetTime(),
346  *previous_data_vel_, *next_data_vel_,
347  this->current_time_, *current_data_vel_);
348 
349  // spatial interpolation
350  this->current_vel_->InterpolateFrom(*current_data_vel_);
351 
352  }
